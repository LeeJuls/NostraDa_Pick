### 4.6 RPC 함수

**place_bet:**
```sql
CREATE OR REPLACE FUNCTION place_bet(
    p_uid UUID, p_issue_id INTEGER, p_choice VARCHAR(3)
) RETURNS JSONB AS $$
DECLARE
    v_deadline TIMESTAMPTZ;
    v_status VARCHAR(10);
BEGIN
    SELECT betting_deadline, status INTO v_deadline, v_status
    FROM Issues WHERE issue_id = p_issue_id
    FOR UPDATE;

    IF v_status IS NULL THEN
        RETURN jsonb_build_object('success', false, 'error', 'issue_not_found');
    END IF;
    IF v_status != 'OPEN' THEN
        RETURN jsonb_build_object('success', false, 'error', 'issue_not_open');
    END IF;
    IF now() > v_deadline THEN
        RETURN jsonb_build_object('success', false, 'error', 'deadline_passed');
    END IF;

    BEGIN
        INSERT INTO Bets (uid, issue_id, choice) VALUES (p_uid, p_issue_id, p_choice);
    EXCEPTION WHEN unique_violation THEN
        RETURN jsonb_build_object('success', false, 'error', 'already_voted');
    END;

    IF p_choice = 'Yes' THEN
        UPDATE Issues SET total_yes = total_yes + 1 WHERE issue_id = p_issue_id;
    ELSE
        UPDATE Issues SET total_no = total_no + 1 WHERE issue_id = p_issue_id;
    END IF;

    UPDATE Users SET
        last_played_date = now(),
        total_played = total_played + 1
    WHERE uid = p_uid;

    RETURN jsonb_build_object('success', true);
END;
$$ LANGUAGE plpgsql;
```

**resolve_issue (벌크):**
```sql
CREATE OR REPLACE FUNCTION resolve_issue(
    p_issue_id INTEGER, p_correct_answer VARCHAR(3)
) RETURNS JSONB AS $$
DECLARE
    v_affected INTEGER;
BEGIN
    UPDATE Issues SET
        correct_answer = p_correct_answer,
        status = 'RESOLVED',
        resolved_at = now()
    WHERE issue_id = p_issue_id AND status = 'CLOSED';

    IF NOT FOUND THEN
        RETURN jsonb_build_object('success', false, 'error', 'issue_not_closable');
    END IF;

    -- Bets에 is_correct 기록
    UPDATE Bets SET is_correct = (choice = p_correct_answer)
    WHERE issue_id = p_issue_id;

    -- 정답 유저: streak +1, wins +1
    UPDATE Users SET
        current_streak = current_streak + 1,
        total_wins = total_wins + 1,
        max_streak = GREATEST(max_streak, current_streak + 1)
    WHERE uid IN (
        SELECT uid FROM Bets
        WHERE issue_id = p_issue_id AND choice = p_correct_answer
    );

    -- 오답 유저: streak = 0
    UPDATE Users SET current_streak = 0
    WHERE uid IN (
        SELECT uid FROM Bets
        WHERE issue_id = p_issue_id AND choice != p_correct_answer
    );

    GET DIAGNOSTICS v_affected = ROW_COUNT;
    RETURN jsonb_build_object('success', true, 'resolved_issue', p_issue_id, 'affected_users', v_affected);
END;
$$ LANGUAGE plpgsql;
```

**delete_issue_cascade:**
```sql
CREATE OR REPLACE FUNCTION delete_issue_cascade(
    p_issue_id INTEGER
) RETURNS JSONB AS $$
DECLARE
    v_status VARCHAR(10);
    v_bet_count INTEGER;
BEGIN
    SELECT status INTO v_status FROM Issues WHERE issue_id = p_issue_id;

    IF v_status IS NULL THEN
        RETURN jsonb_build_object('success', false, 'error', 'not_found');
    END IF;
    IF v_status = 'RESOLVED' THEN
        RETURN jsonb_build_object('success', false, 'error', 'cannot_delete_resolved');
    END IF;

    SELECT COUNT(*) INTO v_bet_count FROM Bets WHERE issue_id = p_issue_id;
    DELETE FROM Issues WHERE issue_id = p_issue_id;

    RETURN jsonb_build_object('success', true, 'deleted_bets', v_bet_count);
END;
$$ LANGUAGE plpgsql;
```

**get_user_rank (내 랭킹 조회):**
```sql
CREATE OR REPLACE FUNCTION get_user_rank(p_uid UUID)
RETURNS JSONB AS $$
DECLARE
    v_rank INTEGER;
    v_total INTEGER;
    v_streak INTEGER;
    v_wins INTEGER;
BEGIN
    SELECT current_streak, total_wins INTO v_streak, v_wins
    FROM Users WHERE uid = p_uid;

    IF NOT FOUND THEN
        RETURN jsonb_build_object('rank', null);
    END IF;

    SELECT COUNT(*) + 1 INTO v_rank
    FROM Users
    WHERE (current_streak > v_streak)
       OR (current_streak = v_streak AND total_wins > v_wins);

    SELECT COUNT(*) INTO v_total FROM Users WHERE total_played > 0;

    RETURN jsonb_build_object(
        'rank', v_rank,
        'total_players', v_total,
        'current_streak', v_streak,
        'total_wins', v_wins
    );
END;
$$ LANGUAGE plpgsql;
```

## 5. 캐싱 전략

> **목적:** 성능 최적화 (응답 속도). Supabase API는 무제한이므로 비용 걱정 불필요.

```python
# services/cache.py
from cachetools import TTLCache
import threading

class CacheManager:
    def __init__(self):
        self._lock = threading.Lock()
        self.open_issues = TTLCache(maxsize=1, ttl=30)      # 30초
        self.rankings = TTLCache(maxsize=1, ttl=60)          # 60초
        self.recent_results = TTLCache(maxsize=1, ttl=30)    # 30초
        self.translations = TTLCache(maxsize=500, ttl=3600)  # 1시간
        self.user_rank = TTLCache(maxsize=200, ttl=60)       # 60초, 유저별

    def get_or_fetch(self, cache, key, fetch_fn):
        with self._lock:
            if key in cache:
                return cache[key]
        result = fetch_fn()
        with self._lock:
            cache[key] = result
        return result

    def invalidate(self, cache, key=None):
        """캐시 무효화. 베팅/결과 확정 후 호출."""
        with self._lock:
            if key:
                cache.pop(key, None)
            else:
                cache.clear()

cache = CacheManager()
```

### 8.3 풀오토 엔진 — 간격 기반 출제

```python
# services/auto_engine.py

def run_auto_cycle():
    """풀오토 1사이클. Render Cron Job으로 매 시간 실행."""
    config = _get_config()
    if config['auto_mode'] != 'on':
        return

    if config.get('auto_engine_lock') == 'true':
        logger.warning("Auto engine already running. Skipping.")
        return

    try:
        _set_lock(True)

        # Step 1: 마감 시간 지난 OPEN → CLOSED
        auto_close_expired()

        # Step 2: CLOSED → RESOLVED (Gemini 판정)
        auto_resolve()

        # Step 3: 새 문제 출제 (간격 기반)
        auto_generate_by_interval(config)

    finally:
        _set_lock(False)


def auto_generate_by_interval(config):
    """
    간격 기반 출제 로직:
    1. OPEN 문제가 min_open 미만이면 → 즉시 출제 (빈화면 방지)
    2. OPEN 문제가 max_open 이상이면 → 출제 안 함
    3. 그 외: 마지막 출제로부터 interval_hours 경과했으면 → 1문제 출제
    """
    open_count = supabase.table('Issues') \
        .select('issue_id', count='exact').eq('status', 'OPEN').execute()
    current_open = open_count.count or 0

    min_open = int(config.get('auto_min_open', 1))
    max_open = int(config.get('auto_max_open', 6))
    interval_hours = int(config.get('auto_interval_hours', 4))
    deadline_hours = int(config.get('auto_deadline_hours', 24))

    # Case 1: 빈화면 방지 — 최소 문제 수 미달
    if current_open < min_open:
        needed = min_open - current_open
        logger.info(f"Below minimum ({current_open} < {min_open}), generating {needed}")
        _generate_issues(needed, deadline_hours)
        return

    # Case 2: 최대치 도달
    if current_open >= max_open:
        logger.info(f"At maximum ({current_open} >= {max_open}), skipping")
        return

    # Case 3: 간격 체크
    last_generated = config.get('last_issue_generated_at', '')
    if last_generated:
        last_time = datetime.fromisoformat(last_generated)
        elapsed = datetime.now(timezone.utc) - last_time
        if elapsed < timedelta(hours=interval_hours):
            logger.info(f"Interval not reached ({elapsed.total_seconds()/3600:.1f}h < {interval_hours}h)")
            return

    # 간격 경과 → 1문제 출제
    logger.info(f"Interval reached, generating 1 issue")
    _generate_issues(1, deadline_hours)


def _generate_issues(count, deadline_hours):
    """Gemini로 문제 출제 + DB 저장"""
    try:
        issues = gemini_generate_issues(count)
        deadline = (
            datetime.now(timezone.utc) +
            timedelta(hours=deadline_hours)
        ).isoformat()

        for issue in issues:
            supabase.table('Issues').insert({
                'category': issue['category'],
                'question_title': issue['question_title'],
                'resolution_criteria': issue['resolution_criteria'],
                'status': 'OPEN',
                'source': 'auto',
                'betting_deadline': deadline
            }).execute()

        # 마지막 출제 시각 기록
        supabase.table('System_Config') \
            .update({
                'value': datetime.now(timezone.utc).isoformat(),
                'updated_at': datetime.now(timezone.utc).isoformat()
            }).eq('key', 'last_issue_generated_at').execute()

        logger.info(f"Generated {len(issues)} issues, deadline: {deadline}")

    except Exception as e:
        logger.error(f"Auto-generate failed: {e}")


def auto_close_expired():
    """betting_deadline 지난 OPEN → CLOSED"""
    result = supabase.table('Issues') \
        .update({'status': 'CLOSED'}) \
        .eq('status', 'OPEN') \
        .lt('betting_deadline', datetime.now(timezone.utc).isoformat()) \
        .execute()
    count = len(result.data) if result.data else 0
    if count:
        logger.info(f"Auto-closed {count} issues")


def auto_resolve():
    """CLOSED → Gemini 판정 → RESOLVED
    [v6.2] resolver_service.py로 분리됨:
    - 48h 초과 이슈: 배치 채점 (50개/청크, 1회 Gemini 호출)
    - 48h 이내 이슈: 개별 채점 (정확도 우선)
    - FALLBACK_MODELS 방식: gemini-3.1-flash-lite-preview 우선
    - key/model rotation: rate limit 시 자동 전환
    GitHub Actions: UTC 0:30, 4:30, 8:30, 12:30, 16:30, 20:30 (6x/day)
    """

    resolver_service.resolve_expired_issues()
```

```python
def translate_issues(issues, lang):
    """이슈를 지정 언어로 번역. Gemini 결과 그대로 사용."""
    if lang == 'en':
        return

    for issue in issues:
        cache_key = f"{issue['issue_id']}_{lang}"
        cached = cache.get_or_fetch(
            cache.translations, cache_key,
            lambda i=issue, l=lang: _fetch_or_translate(i, l)
        )
        if cached:
            issue['question_title'] = cached['translated_title']
            issue['resolution_criteria'] = cached['translated_criteria']

def _fetch_or_translate(issue, lang):
    """DB 캐시 → 없으면 Gemini 번역 → DB 저장. 실패 시 None (영어 유지)."""
    result = supabase.table('Issue_Translations') \
        .select('translated_title, translated_criteria') \
        .eq('issue_id', issue['issue_id']).eq('lang_code', lang) \
        .execute()

    if result.data:
        return result.data[0]

    try:
        translated = gemini_translate(issue['question_title'], issue['resolution_criteria'], lang)
        supabase.table('Issue_Translations').insert({
            'issue_id': issue['issue_id'],
            'lang_code': lang,
            'translated_title': translated['title'],
            'translated_criteria': translated['criteria']
        }).execute()
        return {'translated_title': translated['title'], 'translated_criteria': translated['criteria']}
    except Exception:
        return None
```

## 13. 배포 설정

### render.yaml
```yaml
services:
  - type: web
    name: nostradamus-pick
    runtime: python
    buildCommand: pip install -r requirements.txt
    startCommand: gunicorn app:app --bind 0.0.0.0:$PORT --workers 2
    envVars:
      - key: FLASK_ENV
        value: production
      - key: FLASK_SECRET_KEY
        generateValue: true
      - key: SUPABASE_URL
        sync: false
      - key: SUPABASE_KEY
        sync: false
      - key: GEMINI_API_KEY
        sync: false

  - type: cron
    name: auto-engine
    runtime: python
    buildCommand: pip install -r requirements.txt
    schedule: "0 * * * *"
    startCommand: python -c "from services.auto_engine import run_auto_cycle; run_auto_cycle()"
    envVars:
      - key: SUPABASE_URL
        sync: false
      - key: SUPABASE_KEY
        sync: false
      - key: GEMINI_API_KEY
        sync: false
```

### requirements.txt
```
flask==3.1.0
python-dotenv==1.0.1
supabase==2.11.0
google-generativeai==0.8.4
gunicorn==23.0.0
cachetools==5.5.1
```
